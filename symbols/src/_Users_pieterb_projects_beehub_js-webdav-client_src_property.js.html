 20 // If nl.sara.webdav.Property is already defined, we have a namespace clash!
 21 if (nl.sara.webdav.Property !== undefined) {
 22   throw new nl.sara.webdav.Exception('Class name nl.sara.webdav.Property already taken, could not load JavaScript library for WebDAV connectivity.', nl.sara.webdav.Exception.NAMESPACE_TAKEN);
 23 }
 24 
 25 /**
 26  * @class a WebDAV property
 27  * 
 28  * @param   {Node}  xmlNode  Optionally; the xmlNode describing the propstat object (should be compliant with RFC 4918)
 29  */
 30 nl.sara.webdav.Property = function(xmlNode) {
 31   this._defaultprops = {
 32     'namespace'           : null,
 33     'tagname'             : null,
 34     'value'               : null,
 35     'xmlvalue'            : null,
 36     'status'              : null,
 37     'responsedescription' : null
 38   }
 39   this._errors = [];
 40   
 41   // Constructor logic
 42   if (xmlNode instanceof Node) {
 43     this.set('xmlvalue', xmlNode);  // this also sets value, namespace and tagname
 44   }
 45 }
 46   
 47 /**
 48  * Sets a property
 49  * 
 50  * @param   {String}    prop   The property to update
 51  * @param               value  The value
 52  * @returns  {nl.sara.webdav.Property}         The property itself for chaining methods
 53  */
 54 nl.sara.webdav.Property.prototype.set = function(prop, value) {
 55   if (this._defaultprops[prop] === undefined) {
 56     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
 57   }
 58   switch (prop) {
 59     case 'xmlvalue':
 60       if (!(value instanceof Node)) {
 61         throw new nl.sara.webdav.Exception('xmlvalue must be an instance of Node', nl.sara.webdav.Exception.WRONG_TYPE);
 62       }
 63 
 64       // If we get a new xmlvalue, update the corresponding properties too: value, namespace and tagname
 65       if (value.childNodes.length > 0) {
 66         if ((value.childNodes.item(0).nodeType == 3) || (value.childNodes.item(0).nodeType == 4)) { // Make sure text and CDATA content is stored, even in older browsers
 67           this.set('value', value.childNodes.item(0).nodeValue);
 68         }else{
 69           this.set('value', null);
 70         }
 71       }else{
 72         this.set('value', null);
 73       }
 74       this.set('namespace', value.namespaceURI);
 75       this.set('tagname', value.localName);
 76       break;
 77     case 'value':
 78     case 'namespace':
 79     case 'tagname':
 80       this._defaultprops['xmlvalue'] = null;
 81       break;
 82     case 'status':
 83       value = parseInt(value);
 84       if ((value < 200) || (value >= 600)) {
 85         throw new nl.sara.webdav.Exception('Status must be between 200 and 599 (inclusive)', nl.sara.webdav.Exception.WRONG_VALUE);
 86       }
 87       break;
 88   }
 89   this._defaultprops[prop] = value;
 90   return this;
 91 }
 92 
 93 /**
 94  * Gets a property
 95  * 
 96  * @param   {String}  prop  The property to get
 97  * @returns                  The value of the property
 98  */
 99 nl.sara.webdav.Property.prototype.get = function(prop) {
100   if (this._defaultprops[prop] === undefined) {
101     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
102   }
103   return this._defaultprops[prop];
104 }
105 
106 /**
107  * Adds an error to this property
108  * 
109  * @returns  {Node}  The Node which represents the error
110  */
111 nl.sara.webdav.Property.prototype.addError = function(error) {
112   if (!(error instanceof Node)) {
113     throw new nl.sara.webdav.Exception('Error must be an instance of Node', nl.sara.webdav.Exception.WRONG_TYPE);
114   }
115   this._errors.push(error);
116   return this;
117 }
118 
119 /**
120  * Returns all errors
121  * 
122  * @returns  {array}  An array of Node representing the error
123  */
124 nl.sara.webdav.Property.prototype.getErrors = function() {
125   return this._errors;
126 }
127 
128 /**
129  * Overloads the default toString() method so it returns the value of this property
130  * 
131  * @returns  {String}  A string representation of this property
132  */
133 nl.sara.webdav.Property.prototype.toString = function() {
134   return this.get('value');
135 }
