 21 // If nl.sara.webdav.Property is already defined, we have a namespace clash!
 22 if (nl.sara.webdav.Property !== undefined) {
 23   throw new nl.sara.webdav.Exception('Namespace name nl.sara.webdav.Property already taken, could not load JavaScript library for WebDAV connectivity.', nl.sara.webdav.Exception.NAMESPACE_TAKEN);
 24 }

 26 /**
 27  * @class a WebDAV property
 28  *
 29  * @param  {Node}      xmlNode              Optionally; The xmlNode describing the propstat object (should be compliant with RFC 4918)
 30  * @param  {Number}    status               Optionally; The (HTTP) status code
 31  * @param  {String}    responsedescription  Optionally; The response description
 32  * @param  {String[]}  errors               Optionally; An array of errors
 33  * @property  {String}    namespace            The namespace
 34  * @property  {String}    tagname              The tag name
 35  * @property  {NodeList}  xmlvalue             A NodeList with the value of this property
 36  * @property  {String}    value                A textual representation of xmlvalue
 37  * @property  {Number}    status               The (HTTP) status code
 38  * @property  {String}    responsedescription  The response description
 39  */
 40 nl.sara.webdav.Property = function(xmlNode, status, responsedescription, errors) {
 41   // First define private attributes
 42   Object.defineProperty(this, '_namespace', {
 43     'value': null,
 44     'enumerable': false,
 45     'configurable': false,
 46     'writable': true
 47   });
 48   Object.defineProperty(this, '_tagname', {
 49     'value': null,
 50     'enumerable': false,
 51     'configurable': false,
 52     'writable': true
 53   });
 54   Object.defineProperty(this, '_value', {
 55     'value': null,
 56     'enumerable': false,
 57     'configurable': false,
 58     'writable': true
 59   });
 60   Object.defineProperty(this, '_xmlvalue', {
 61     'value': null,
 62     'enumerable': false,
 63     'configurable': false,
 64     'writable': true
 65   });
 66   Object.defineProperty(this, '_status', {
 67     'value': null,
 68     'enumerable': false,
 69     'configurable': false,
 70     'writable': true
 71   });
 72   Object.defineProperty(this, '_responsedescription', {
 73     'value': null,
 74     'enumerable': false,
 75     'configurable': false,
 76     'writable': true
 77   });
 78   Object.defineProperty(this, '_errors', {
 79     'value': [],
 80     'enumerable': false,
 81     'configurable': false,
 82     'writable': true
 83   });
 84 
 85   // Constructor logic
 86   if (xmlNode instanceof Node) {
 87     this.namespace = xmlNode.namespaceURI;
 88     this.tagname = xmlNode.localName;
 89     this.xmlvalue = xmlNode.childNodes;
 90   }
 91   if (status !== undefined) {
 92     this.status = status;
 93   }
 94   if (responsedescription !== undefined) {
 95     this.responsedescription = responsedescription;
 96   }
 97   if (errors instanceof Array) {
 98     for (var i = 0; i < errors.length; i++) {
 99       this.addError(errors[i]);
100     }
101   }
102 }
103

123 Object.defineProperty(nl.sara.webdav.Property.prototype, 'value', {
124   'set': function(value) {
125     this._xmlvalue = null;
126     this._value = value;
127   },
128   'get': function() {
129     return this._value;
130   }
131 });
132 
133 Object.defineProperty(nl.sara.webdav.Property.prototype, 'xmlvalue', {
134   'set': function(value) {
135     if (value === null) {
136       this._value = null;
137       this._xmlvalue = null;
138       return;
139     }
140     if (!(value instanceof NodeList)) {
141       throw new nl.sara.webdav.Exception('xmlvalue must be an instance of NodeList', nl.sara.webdav.Exception.WRONG_TYPE);
142     }
143 
144     // TODO: handle convertion of value with codecs
145     if (value.length > 0) {
146       this._value = '';
147       for (var i = 0; i < value.length; i++) {
148         var node = value.item(i);
149         if ((node.nodeType == 3) || (node.nodeType == 4)) { // Make sure text and CDATA content is stored
150           this._value += node.nodeValue;
151         }else{ // If even one of the nodes is not text or CDATA, then we don't parse a text value at all
152           this._value = null;
153           break;
154         }
155       }
156     }else{
157       this._value = null;
158     }
159     this._xmlvalue = value;
160   },
161   'get': function() {
162     return this._xmlvalue;
163   }
164 });
165 
166 Object.defineProperty(nl.sara.webdav.Property.prototype, 'status', {
167   'set': function(value) {
168     var status = parseInt(value, 10);
169     if ((status < 200) || (status >= 600)) {
170       throw new nl.sara.webdav.Exception('Status must be between 200 and 599 (inclusive)', nl.sara.webdav.Exception.WRONG_VALUE);
171     }
172     this._status = status;
173   },
174   'get': function() {
175     return this._status;
176   }
177 });
178 
179 Object.defineProperty(nl.sara.webdav.Property.prototype, 'responsedescription', {
180   'set': function(value) {
181     this._responsedescription = value;
182   },
183   'get': function() {
184     return this._responsedescription;
185   }
186 });

188 //########################## DEFINE PUBLIC METHODS #############################
189 /**
190 * Adds an error to this property
191 *
192 * @returns {Node} The Node which represents the error
193 */
194 nl.sara.webdav.Property.prototype.addError = function(error) {
195   if (!(error instanceof Node)) {
196     throw new nl.sara.webdav.Exception('Error must be an instance of Node', nl.sara.webdav.Exception.WRONG_TYPE);
197   }
198   this._errors.push(error);
199   return this;
200 }
201 
202 /**
203 * Returns all errors
204 *
205 * @returns {array} An array of Node representing the error
206 */
207 nl.sara.webdav.Property.prototype.getErrors = function() {
208   return this._errors;
209 }
