 21 /**
 22  * This plugin adds acl capabilities to the WebDAV client class

 25 /**
 26  * Perform a WebDAV ACL request
 27  *
 28  * @param    {String}                        path      The path to perform ACL on
 29  * @param    {Function(status,Multistatus)}  callback  Querying the server is done asynchronously, this callback function is called when the results are in
 30  * @param    {nl.sara.webdav.Acl}            acl       The ACL to submit
 31  * @returns  {nl.sara.webdav.Client}                   The client itself for chaining methods
 32  */
 33 nl.sara.webdav.Client.prototype.acl = function(path, callback, acl) {
 34   if ((path === undefined) || (callback === undefined)) {
 35     throw new nl.sara.webdav.Exception('ACL requires the parameters path, callback and acl', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
 36   }
 37   if (!(typeof path == "string") || !(acl instanceof nl.sara.webdav.Acl)) {
 38     throw new nl.sara.webdav.Exception('ACL parameter; path should be a string, acl should be an instance of Acl', nl.sara.webdav.Exception.WRONG_TYPE);
 39   }
 40 
 41   // Get the full URL, based on the specified path
 42   var url = this.getUrl(path);
 43 
 44   var aclBody = document.implementation.createDocument("DAV:", "acl", null);
 45   var aclLength = acl.getLength();
 46   for (var i = 0; i < aclLength; i++) { // Loop over the ACE's in this ACL
 47     var ace = acl.getAce(i);
 48     var aceBody = aclBody.createElementNS('DAV:', 'ace');
 49 
 50     // First create a principal node
 51     var principal = aclBody.createElementNS('DAV:', 'principal');
 52     var princVal = ace.principal;
 53     switch (princVal) {
 54       case nl.sara.webdav.Ace.ALL:
 55         principal.appendChild(aclBody.createElementNS('DAV:', 'all'));
 56         break;
 57       case nl.sara.webdav.Ace.AUTHENTICATED:
 58         principal.appendChild(aclBody.createElementNS('DAV:', 'authenticated'));
 59         break;
 60       case nl.sara.webdav.Ace.UNAUTHENTICATED:
 61         principal.appendChild(aclBody.createElementNS('DAV:', 'unauthenticated'));
 62         break;
 63       case nl.sara.webdav.Ace.SELF:
 64         principal.appendChild(aclBody.createElementNS('DAV:', 'self'));
 65         break;
 66       default: // If it isn't one of the constants, it should be either a Property object or a string/URL
 67         if (!(princVal instanceof nl.sara.webdav.Property)) { // It is a string; the URL of the principal
 68           var href = aclBody.createElementNS('DAV:', 'href');
 69           href.appendChild(aclBody.createCDATASection(princVal));
 70           principal.appendChild(href);
 71         }else{ // And else it is a property
 72           var property = aclBody.createElementNS('DAV:', 'property');
 73           var prop = aclBody.createElementNS(princVal.namespace, princVal.tagname);
 74           if (princVal.xmlvalue != null) {
 75             for (var j = 0; j < princVal.xmlvalue.length; j++) {
 76               prop.appendChild(princVal.xmlValue.item(j));
 77             }
 78           }
 79           property.appendChild(prop);
 80           principal.appendChild(property);
 81         }
 82       break;
 83     }
 84 
 85     // If the principal should be inverted, put it in an 'invert' element
 86     if (ace.invertprincipal) {
 87       var invert = aclBody.createElementNS('DAV:', 'invert');
 88       invert.appendChild(principal);
 89       aceBody.appendChild(invert);
 90     }else{
 91       aceBody.appendChild(principal);
 92     }
 93 
 94     // Then prepare the privileges
 95     // grant or deny?
 96     var privilegeParent = null;
 97     if (ace.grantdeny == nl.sara.webdav.Ace.DENY) {
 98       privilegeParent = aceBody.createElementNS('DAV:', 'deny');
 99     }else if (ace.grantdeny == nl.sara.webdav.Ace.GRANT) {
100       privilegeParent = aceBody.createElementNS('DAV:', 'grant');
101     }else{
102       throw new nl.sara.webdav.Exception('\'grantdeny\' property not set on one of the ACE\'s in this ACL', nl.sara.webdav.Exception.WRONG_VALUE);
103     }
104     var namespaces = ace.getNamespaceNames();
105     for (var j = 0; j < namespaces.length; j++) { // loop through namespaces
106       var namespace = namespaces[j];
107       var privileges = ace.getPrivilegeNames(namespace);
108       for (var k = 0; k < privileges.length; k++) { // loop through each privilege in this namespace
109         var privilege = privileges[k];
110         var privilegeElement = aclBody.createElementNS('DAV:', 'privilege');
111         var priv = aclBody.createElementNS(privilege.namespace, privilege.tagname);
112         if (privilege.xmlvalue != null) {
113           for (var l = 0; l < privilege.xmlvalue.length; l++) {
114             priv.appendChild(privilege.xmlValue.item(j));
115           }
116         }
117         privilegeElement.appendChild(priv);
118         privilegeParent.appendChild(privilegeElement);
119       }
120     }
121     aceBody.appendChild(privilegeParent);
122 
123     aclBody.documentElement.appendChild(aceBody);
124   }
125 
126   // Create the request body string
127   var serializer = new XMLSerializer();
128   var body = '<?xml version="1.0" encoding="utf-8" ?>' + serializer.serializeToString(aclBody);
129 
130   // And then send the request
131   var ajax = nl.sara.webdav.Client.getAjax("ACL", url, callback);
132   ajax.setRequestHeader('Content-Type', 'application/xml; charset="utf-8"');
133   ajax.send(body);
134 
135   return this;
136 }

138 /**
139  * Perform a WebDAV REPORT request
140  *
141  * @param    {String}                        path         The path to perform REPORT on
142  * @param    {Function(status,Multistatus)}  callback     Querying the server is done asynchronously, this callback function is called when the results are in
143  * @param    {Document}                      body         The (XML DOM) document to parse and send as the request body
144  * @returns  {nl.sara.webdav.Client}                      The client itself for chaining methods
145  */
146 nl.sara.webdav.Client.prototype.report = function(path, callback, body) {
147   if ((path === undefined) || (callback === undefined) || (body === undefined)) {
148     throw new nl.sara.webdav.Exception('POST requires the parameters path, callback and body', nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER);
149   }
150   if ((typeof path != "string") || (typeof body != 'string') || !(body instanceof Document)) {
151     throw new nl.sara.webdav.Exception('POST parameter; path should be a string, body should be an instance of Document', nl.sara.webdav.Exception.WRONG_TYPE);
152   }
153 
154   // Get the full URL, based on the specified path
155   var url = this.getUrl(path);
156 
157   // Parse the body
158   var serializer = new XMLSerializer();
159   var body = '<?xml version="1.0" encoding="utf-8" ?>' + serializer.serializeToString(body);
160 
161   // And then send the request
162   var ajax = null;
163   ajax = nl.sara.webdav.Client.getAjax("REPORT", url, callback);
164   ajax.setRequestHeader('Content-Type', 'application/xml; charset="utf-8"');
165   ajax.send(body);
166 
167   return this;
168 }
