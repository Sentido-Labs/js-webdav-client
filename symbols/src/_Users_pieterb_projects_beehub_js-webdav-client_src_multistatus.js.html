 20 // If nl.sara.webdav.Multistatus is already defined, we have a namespace clash!
 21 if (nl.sara.webdav.Multistatus !== undefined) {
 22   throw new nl.sara.webdav.Exception('Namespace nl.sara.webdav.Multistatus already taken, could not load JavaScript library for WebDAV connectivity.', nl.sara.webdav.Exception.NAMESPACE_TAKEN);
 23 }

 25 /**
 26  * @class WebDAV Multistatus response body
 27  * 
 28  * @param   {Node}  xmlNode  Optionally; the xmlNode describing the multistatus object (should be compliant with RFC 4918)
 29  */
 30 nl.sara.webdav.Multistatus = function(xmlNode) {
 31   this._defaultprops = {
 32     'responsedescription' : null
 33   }
 34   this._responses = {};
 35   
 36   // Constructor logic
 37   if (xmlNode instanceof Node) { 
 38     if ((xmlNode.namespaceURI.toLowerCase() != 'dav:') || (xmlNode.localName.toLowerCase() != 'multistatus')) {
 39       throw new nl.sara.webdav.Exception('Node is not of type DAV:multistatus', nl.sara.webdav.Exception.WRONG_XML);
 40     }
 41     var data = xmlNode.childNodes;
 42     for (var i = 0; i < data.length; i++) {
 43       var child = data.item(i);
 44       if ((child.namespaceURI == null) || (child.namespaceURI.toLowerCase() != 'dav:')) { // Skip if not from the right namespace
 45         continue;
 46       }
 47       switch (child.localName) {
 48         case 'responsedescription': // responsedescription is always CDATA, so just take the text
 49           this.set('responsedescription', child.childNodes.item(0).nodeValue);
 50           break;
 51         case 'response': // response node should be parsed further
 52           var response = new nl.sara.webdav.Response(child);
 53           var responseChilds = child.childNodes;
 54           var hrefs = [];
 55           for (var j = 0; j < responseChilds.length; j++) {
 56             var responseChild = responseChilds.item(j);
 57             if ((responseChild.localName == 'href') && (responseChild.namespaceURI != null) && (responseChild.namespaceURI.toLowerCase() == 'dav:')) { // For each HREF element we create a separate response object
 58               hrefs.push(responseChild.childNodes.item(0).nodeValue);
 59             }
 60           }
 61           if (hrefs.length > 1) { // Multiple HREFs = start copying the response (this makes sure it is not parsed over and over again). No deep copying needed; there can't be a propstat
 62             for (var k = 0; k < hrefs.length; k++) {
 63               var copyResponse = new nl.sara.webdav.Response();
 64               copyResponse.set('href', hrefs[k]);
 65               copyResponse.set('status', response.get('status'));
 66               copyResponse.set('error', response.get('error'));
 67               copyResponse.set('responsedescription', response.get('responsedescription'));
 68               copyResponse.set('location', response.get('location'));
 69               this.addResponse(copyResponse);
 70             }
 71           }else{
 72             this.addResponse(response);
 73           }
 74         break;
 75       }
 76     }
 77   }
 78 }

 80 /**
 81  * Adds a Response
 82  * 
 83  * @param   {nl.sara.webdav.Response}     response  The response
 84  * @returns  {nl.sara.webdav.Multistatus}            The multistatus itself for chaining methods
 85  */
 86 nl.sara.webdav.Multistatus.prototype.addResponse = function(response) {
 87   if (!(response instanceof nl.sara.webdav.Response)) {
 88     throw new nl.sara.webdav.Exception('Response should be instance of Response', nl.sara.webdav.Exception.WRONG_TYPE);
 89   }
 90   var name = response.get('href');
 91   while (name.substring(name.length-1) == '/') {
 92     name = name.substr(0, name.length-1);
 93   }
 94   name = name.substr(name.lastIndexOf('/')+1);
 95   this._responses[name] = response;
 96   return this;
 97 }

 99 /**
100  * Gets a Response
101  * 
102  * @param   {String}    name  The name of the response to get
103  * @returns  {nl.sara.webdav.Response}        The value of the WebDAV property or undefined if the WebDAV property doesn't exist
104  */
105 nl.sara.webdav.Multistatus.prototype.getResponse = function(name) {
106   return this._responses[name];
107 }
108 
109 /**
110  * Gets the response names
111  * 
112  * @returns  {String[]}  The names of the different responses
113  */
114 nl.sara.webdav.Multistatus.prototype.getResponseNames = function() {
115   return Object.keys(this._responses);
116 }
117 
118 /**
119  * Sets a property
120  * 
121  * @param   {String}       prop   The property to update
122  * @param                  value  The value
123  * @returns  {nl.sara.webdav.Multistatus}         The multistatus object itself for chaining methods
124  */
125 nl.sara.webdav.Multistatus.prototype.set = function(prop, value) {
126   if (this._defaultprops[prop] === undefined) {
127     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
128   }
129   this._defaultprops[prop] = value;
130   return this;
131 }
132 
133 /**
134  * Gets a property
135  * 
136  * @param   {String}  prop  The property to get
137  * @returns                  The value of the property or undefined if the property doesn't exist
138  */
139 nl.sara.webdav.Multistatus.prototype.get = function(prop) {
140   if (this._defaultprops[prop] === undefined) {
141     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
142   }
143   return this._defaultprops[prop];
144 }
