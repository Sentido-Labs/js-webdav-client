 20 // If nl.sara.webdav.Privilege is already defined, we have a namespace clash!
 21 if (nl.sara.webdav.Privilege !== undefined) {
 22   throw new nl.sara.webdav.Exception('Namespace nl.sara.webdav.Privilege already taken, could not load JavaScript library for WebDAV connectivity.', nl.sara.webdav.Exception.NAMESPACE_TAKEN);
 23 }
 24 
 25 /**
 26  * @class WebDAV ACL privilege
 27  * 
 28  * @param   {Node}  xmlNode  Optionally; the xmlNode describing the privilege object
 29  */
 30 nl.sara.webdav.Privilege = function(xmlNode) {
 31   this._defaultprops = {
 32     'namespace'           : null,
 33     'tagname'             : null,
 34     'value'               : null,
 35     'xmlvalue'            : null
 36   }
 37   
 38   // Constructor logic
 39   if (xmlNode instanceof Node) {
 40     this.set('xmlvalue', xmlNode);
 41   }
 42 }
 43   
 44 /**
 45  * Sets a property
 46  * 
 47  * @param   {String}    prop   The property to update
 48  * @param               value  The value
 49  * @returns  {nl.sara.webdav.Privilege}        The privilege itself for chaining methods
 50  */
 51 nl.sara.webdav.Privilege.prototype.set = function(prop, value) {
 52   if (this._defaultprops[prop] === undefined) {
 53     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
 54   }
 55   switch (prop) {
 56     case 'xmlvalue':
 57       if (!(value instanceof Node)) {
 58         throw new nl.sara.webdav.Exception('xmlvalue must be an instance of Node', nl.sara.webdav.Exception.WRONG_TYPE);
 59       }
 60 
 61       // If we get a new xmlvalue, update the corresponding properties too: value, namespace and tagname
 62       if (value.childNodes.length > 0) {
 63         if ((value.childNodes.item(0).nodeType == 3) || (value.childNodes.item(0).nodeType == 4)) { // Make sure text and CDATA content is stored, even in older browsers
 64           this.set('value', value.childNodes.item(0).nodeValue);
 65         }else if (value.textContent) { // For other type of nodes we just hope this will do something useful, although you probably want to examine the xmlvalue yourself anyway.
 66           this.set('value', value.textContent); // This could fail in older browsers, but that's no problem. See comments in the line above.
 67         }else{
 68           this.set('value', null);
 69         }
 70       }else{
 71         this.set('value', null);
 72       }
 73       this.set('namespace', value.namespaceURI);
 74       this.set('tagname', value.localName);
 75       break;
 76     case 'value':
 77       this._defaultprops['xmlvalue'] = null;
 78     break;
 79   }
 80   this._defaultprops[prop] = value;
 81   return this;
 82 }
 83 
 84 /**
 85  * Gets a property
 86  * 
 87  * @param   {String}  prop  The property to get
 88  * @returns                  The value of the property
 89  */
 90 nl.sara.webdav.Privilege.prototype.get = function(prop) {
 91   if (this._defaultprops[prop] === undefined) {
 92     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
 93   }
 94   return this._defaultprops[prop];
 95 }
 96 
 97 /**
 98  * Overloads the default toString() method so it returns the value of this privilege
 99  * 
100  * @returns  {String}  A string representation of this privilege
101  */
102 nl.sara.webdav.Privilege.prototype.toString = function() {
103   return this.get('value');
104 }
