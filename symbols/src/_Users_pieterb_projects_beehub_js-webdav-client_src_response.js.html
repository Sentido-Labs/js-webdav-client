 20 // If nl.sara.webdav.Response is already defined, we have a namespace clash!
 21 if (nl.sara.webdav.Response !== undefined) {
 22   throw new nl.sara.webdav.Exception('Namespace nl.sara.webdav.Response already taken, could not load JavaScript library for WebDAV connectivity.', nl.sara.webdav.Exception.NAMESPACE_TAKEN);
 23 }
 24 
 25 /**
 26  * @class a WebDAV response
 27  * 
 28  * @param   {Node}  xmlNode  Optionally; the xmlNode describing the response object (should be compliant with RFC 4918)
 29  */
 30 nl.sara.webdav.Response = function(xmlNode) {
 31   this._namespaces = {};
 32   this._defaultprops = {
 33     'href'                : null,
 34     'status'              : null,
 35     'error'               : null,
 36     'responsedescription' : null,
 37     'location'            : null
 38   }
 39   
 40   // Constructor logic
 41   if (xmlNode instanceof Node) { 
 42     if ((xmlNode.namespaceURI.toLowerCase() != 'dav:') || (xmlNode.localName.toLowerCase() != 'response')) {
 43       throw new nl.sara.webdav.Exception('Node is not of type DAV:response', nl.sara.webdav.Exception.WRONG_XML);
 44     }
 45     var data = xmlNode.childNodes;
 46     for (var i = 0; i < data.length; i++) {
 47       var child = data.item(i);
 48       if ((child.namespaceURI == null) || (child.namespaceURI.toLowerCase() != 'dav:')) { // Skip if not from the right namespace
 49         continue;
 50       }
 51       switch (child.localName) {
 52         case 'href':
 53         case 'status':
 54         case 'error':
 55         case 'responsedescription':
 56           // always CDATA, so just take the text
 57           this.set(child.localName, child.childNodes.item(0).nodeValue);
 58           break;
 59         case 'location':
 60           this.set(child.localName, child.childNodes.item(0).childNodes.item(0).nodeValue);
 61           break;
 62         case 'propstat': // propstat node should be parsed further
 63           var propstatChilds = child.childNodes;
 64           // First find status, error, responsedescription and props (as Node objects, not yet as Property objects)
 65           var status = null;
 66           var errors = [];
 67           var responsedescription = null;
 68           var props = [];
 69           for (var j = 0; j < propstatChilds.length; j++) { // Parse the child nodes of the propstat element
 70             var propstatChild = propstatChilds.item(j);
 71             if ((propstatChild.nodeValue != 1) || (propstatChild.namespaceURI != 'DAV:')) {
 72               continue;
 73             }
 74             switch (propstatChild.localName) {
 75               case 'prop':
 76                 for (var k = 0; k < propstatChild.childNodes.length; k++) {
 77                   props.push(propstatChild.childNodes.item(k));
 78                 }
 79                 break;
 80               case 'error':
 81                 for (var k = 0; k < propstatChild.childNodes.length; k++) {
 82                   errors.push(propstatChild.childNodes.item(k));
 83                 }
 84                 break;
 85                 break;
 86               case 'status': // always CDATA, so just take the text
 87                 status = child.childNodes.item(0).nodeValue;
 88                 break;
 89               case 'responsedescription': // always CDATA, so just take the text
 90                 responsedescription = child.childNodes.item(0).nodeValue;
 91               break;
 92             }
 93           }
 94           
 95           // Then create and add a new property for each element found in DAV:prop
 96           for (var j = 0; j < props.length; j++) {
 97             var property = new nl.sara.webdav.Property();
 98             property.set('xmlvalue', props[j]);
 99             property.set('status', status);
100             property.set('responsedescription', responsedescription);
101             for (var k = 0; k < errors.length; k++) {
102               property.addError(errors[k]);
103             }
104             this.addProperty(property);
105           }
106         break;
107       }
108     }
109   }
110 }

112 /**
113  * Adds a WebDAV property
114  * 
115  * @param   {nl.sara.webdav.Property}  property  The property
116  * @returns  {nl.sara.webdav.Response}            The response itself for chaining methods
117  */
118 nl.sara.webdav.Response.prototype.addProperty = function(property) {
119   if (!(property instanceof nl.sara.webdav.Property)) {
120     throw new nl.sara.webdav.Exception('Response property should be instance of Property', nl.sara.webdav.Exception.WRONG_TYPE);
121   }
122   var namespace = property.get('namespace');
123   if (namespace) {
124     if (this._namespaces[namespace] === undefined) {
125       this._namespaces[namespace] = {};
126     }
127   }else{
128     throw new nl.sara.webdav.Exception('Response property should have a namespace', nl.sara.webdav.Exception.WRONG_TYPE);
129   }
130   
131   this._namespaces[namespace][property.get('tagname')] = property;
132   return this;
133 }
134 
135 /**
136  * Gets a WebDAV property
137  * 
138  * @param   {String}    namespace  The namespace of the WebDAV property
139  * @param   {String}    prop       The WebDAV property to get
140  * @returns  {nl.sara.webdav.Property}             The value of the WebDAV property or undefined if the WebDAV property doesn't exist
141  */
142 nl.sara.webdav.Response.prototype.getProperty = function(namespace, prop) {
143   if ((this._namespaces[namespace] === undefined) || (this._namespaces[namespace][prop] === undefined)) {
144     return undefined;
145   }
146   return this._namespaces[namespace][prop];
147 }
148 
149 /**
150  * Gets the namespace names
151  * 
152  * @returns  {String[]}  The names of the different namespaces
153  */
154 nl.sara.webdav.Response.prototype.getNamespaceNames = function() {
155   return Object.keys(this._namespaces);
156 }
157 
158 /**
159  * Gets the property names of a namespace
160  * 
161  * @param   {String}    namespace  The namespace of the WebDAV property
162  * @returns  {String[]}             The names of the different properties of a namespace
163  */
164 nl.sara.webdav.Response.prototype.getPropertyNames = function(namespace) {
165   if (this._namespaces[namespace] === undefined) {
166     return new Array();
167   }
168   return Object.keys(this._namespaces[namespace]);
169 }
170 
171 /**
172  * Sets a property
173  * 
174  * @param   {String}    prop   The property to update
175  * @param               value  The value
176  * @returns  {nl.sara.webdav.Response}         The response itself for chaining methods
177  */
178 nl.sara.webdav.Response.prototype.set = function(prop, value) {
179   if (this._defaultprops[prop] === undefined) {
180     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
181   }
182   this._defaultprops[prop] = value;
183   return this;
184 }
185 
186 /**
187  * Gets a property
188  * 
189  * @param   {String}  prop  The property to get
190  * @returns                  The value of the property
191  */
192 nl.sara.webdav.Response.prototype.get = function(prop) {
193   if (this._defaultprops[prop] === undefined) {
194     throw new nl.sara.webdav.Exception('Property ' + prop + ' does not exist', nl.sara.webdav.Exception.UNEXISTING_PROPERTY);
195   }
196   return this._defaultprops[prop];
197 }
