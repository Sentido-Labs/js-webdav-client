 20 // If nl.sara.webdav.Exception is already defined, we have a namespace clash!
 21 if (nl.sara.webdav.Exception !== undefined) {
 22   throw 'Class name nl.sara.webdav.Exception already taken, could not load JavaScript library for WebDAV connectivity.';
 23 }
 24 
 25 /**
 26  * @class an exception
 27  * @param  {String}  message  A human readable message
 28  * @param  {Number}     code     The error code. It is best to use the class constants to set this.
 29  */
 30 nl.sara.webdav.Exception = function(message, code) {
 31   this._message = null;
 32   this._code = null;
 33   
 34   // Constructor logic
 35   if (message !== undefined) {
 36     this.setMessage(message);
 37   }
 38   if (code !== undefined) {
 39     this.setCode(code);
 40   }
 41 }
 42 
 43 /**#@+
 44  * Declaration of the error code constants
 45  */
 46 nl.sara.webdav.Exception.WRONG_TYPE = 1;
 47 nl.sara.webdav.Exception.NAMESPACE_TAKEN = 2;
 48 nl.sara.webdav.Exception.UNEXISTING_PROPERTY = 3;
 49 nl.sara.webdav.Exception.WRONG_XML = 4;
 50 nl.sara.webdav.Exception.WRONG_VALUE = 5;
 51 nl.sara.webdav.Exception.MISSING_REQUIRED_PARAMETER = 6;
 52 nl.sara.webdav.Exception.AJAX_ERROR = 7;
 53 nl.sara.webdav.Exception.NOT_IMPLEMENTED = 8;
 54 /**#@-*/
 55 
 56 /**
 57  * Sets the message
 58  * 
 59  * @param   {String}  message  The message
 60  * @returns  {nl.sara.webdav.Exception}        self
 61  */
 62 nl.sara.webdav.Exception.prototype.setMessage = function(message) {
 63   this._message = message;
 64   return this;
 65 }
 66 
 67 /**
 68  * Gets the message
 69  * 
 70  * @returns           The message
 71  */
 72 nl.sara.webdav.Exception.prototype.getMessage = function() {
 73   return this._message;
 74 }
 75 
 76 /**
 77  * Sets the code
 78  * 
 79  * @param   {Number}        code  The code
 80  * @returns  {nl.sara.webdav.Exception}        self
 81  */
 82 nl.sara.webdav.Exception.prototype.setCode = function(code) {
 83   if (isNaN(code)) {
 84     throw new nl.sara.webdav.Exception('nl.sara.webdav.Exception.setCode(code) expects an integer', nl.sara.webdav.Exception.WRONG_TYPE);
 85   }
 86   this._code = parseInt(code, 10);
 87   return this;
 88 }
 89 
 90 /**
 91  * Gets the code
 92  * 
 93  * @returns  {Number}  The code
 94  */
 95 nl.sara.webdav.Exception.prototype.getCode = function() {
 96   return this._code;
 97 }
 98     
 99 /**
100  * Overloads the default toString() method so it returns the message of this exception
101  * 
102  * @returns  {String}  A string representation of this property
103  */
104 nl.sara.webdav.Exception.prototype.toString = function() {
105   return this.getMessage();
106 }
